            <p id="b" class="invalid">Atleast one number or more</p>
            <p id="c" class="invalid">Atleast one syllabal</p>

<script>
     var myInput = document.getElementById('pwd');
     var highLetterCheck = document.getElementById('a');
     var oneNumber = document.getElementById('b');
     var syllabelCheck = document.getElementById('c');
     var numberOfChar = document.getElementById('d');
     var lowerCase = document.getElementById('e');

     myInput.onkeyup = function() {
    var higherCaseLetters= /[A-Z]/g;
    if (myInput.value.match(higherCaseLetters)){
        highLetterCheck.classList.remove('invalid');
        highLetterCheck.classList.add('valid'); 
    }else{
        highLetterCheck.classList.remove('valid');
        highLetterCheck.classList.add('invalid');
    } 
     
    var numbers = /[0-9]/g;
    if (myInput.value.match(numbers)){
        oneNumber.classList.remove('invalid');
        oneNumber.classList.add('valid');
    }
    else{
        oneNumber.classList.remove('valid');
        oneNumber.classList.add('invalid');
    }
    
    var symbols = /[ @ # $ % ^ & * )]/;
    if(myInput.value.match(symbols)){
        syllabelCheck.classList.remove('invalid');
        syllabelCheck.classList.add('valid');
    }else{
        syllabelCheck.classList.remove('valid');
        syllabelCheck.classList.add('invalid'); 
    }
       
    if(myInput.value.length >= 6){
        numberOfChar.classList.remove('invalid');
        numberOfChar.classList.add('valid');
    }else{
        numberOfChar.classList.remove('valid');
        numberOfChar.classList.add('invalid');
    }
     
    var lowerLetters = /[a-z]/g;
    if(myInput.value.match(lowerLetters)){
        lowerCase.classList.remove('invalid');
        lowerCase.classList.add('valid');
    }else {
        lowerCase.classList.add('invalid');
        lowerCase.classList.remove('valid');
    }  
} 
function showPassword() {
        var tog = document.getElementById('pwd');
        if (tog.type === "password"){
           tog.type = "text";
        }
        else{
            tog.type ="password";
        }
    }
